mostrarVistaPreviaYImprimir(tarjeta: any): void {
  const vistaPreviaHtml = this.generarVistaPreviaImpresion(tarjeta);
  const ventana = window.open('', 'PRINT', 'width=600,height=400');
  ventana.document.write(vistaPreviaHtml);
  ventana.document.close(); // Necesario para algunos navegadores
  ventana.focus();

  // Opcional: Desencadenar la impresión automáticamente después de mostrar la vista previa
  ventana.onload = () => {
    ventana.print();
    ventana.close();
  };
}
