import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-mi-componente',
  templateUrl: './mi-componente.component.html',
  styleUrls: ['./mi-componente.component.css']
})
export class MiComponenteComponent implements OnInit {
  cadenaNombresApellidos: string = "Juan  Carlos    Pérez  López";
  nombres: string = '';
  apellidos: string = '';
  fechaVencimiento: string = "12/24"; // Ejemplo

  ngOnInit(): void {
    this.procesarNombresApellidos();
  }

  procesarNombresApellidos(): void {
    // Transforma dos o más espacios en uno solo
    let cadenaLimpia = this.cadenaNombresApellidos.replace(/\s+/g, ' ');
    let partes = cadenaLimpia.split(' ');

    if (partes.length >= 4) {
      this.nombres = `${partes[0]} ${partes[1]}`;
      this.apellidos = `${partes[2]} ${partes.slice(3).join(' ')}`;
    } else if (partes.length === 3) {
      this.nombres = partes[0];
      this.apellidos = `${partes[1]} ${partes[2]}`;
    } else if (partes.length === 2) {
      this.nombres = partes[0];
      this.apellidos = partes[1];
    } else {
      // Ajusta según necesidad, por ejemplo, para manejar un solo nombre sin apellidos.
      this.nombres = partes[0];
    }
  }
}
