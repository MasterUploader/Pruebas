public static string FormatRequestInfo(string method, string url, string requestBody)
{
    string formattedJson = string.IsNullOrWhiteSpace(requestBody) 
        ? "        (Sin cuerpo en la solicitud)"
        : FormatJson(requestBody, 8); // Aplica indentación controlada

    return $@"
----------------------------------Request Info---------------------------------
Inicio: {DateTime.Now}
-------------------------------------------------------------------------------
Método: {method}
URL: {url}
Cuerpo:
{formattedJson}
----------------------------------Request Info---------------------------------
Fin: {DateTime.Now}
-------------------------------------------------------------------------------";
}


public static string FormatResponseInfo(int statusCode, string responseBody)
{
    string formattedJson = string.IsNullOrWhiteSpace(responseBody) 
        ? "        (Sin cuerpo en la respuesta)"
        : FormatJson(responseBody, 8); // Aplica indentación controlada

    return $@"
----------------------------------Response Info---------------------------------
Inicio: {DateTime.Now}
-------------------------------------------------------------------------------
Código Estado: {statusCode}
Cuerpo:
{formattedJson}
----------------------------------Response Info---------------------------------
Fin: {DateTime.Now}
-------------------------------------------------------------------------------";
}
