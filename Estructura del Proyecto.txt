public static string FormatJson(string json)
{
    try
    {
        using var jsonDoc = JsonDocument.Parse(json);
        return JsonSerializer.Serialize(jsonDoc, new JsonSerializerOptions
        {
            WriteIndented = true,   // Indentación para mejorar legibilidad
            Encoder = System.Text.Encodings.Web.JavaScriptEncoder.UnsafeRelaxedJsonEscaping // Permite caracteres Unicode sin escape
        });
    }
    catch
    {
        return json; // Si falla la conversión, devolver el JSON original
    }
}
